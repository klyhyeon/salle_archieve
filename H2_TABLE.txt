//H2 SQL

//PRODUCT TABLE
CREATE TABLE PRODUCT (
  PR_ID BIGINT AUTO_INCREMENT, 
  PR_EMAIL VARCHAR2(50) NOT NULL,
  PR_REG_DATE TIMESTAMP WITH TIME ZONE,
  PR_IMG_1 VARCHAR2(50) NOT NULL, 
  PR_IMG_2 VARCHAR2(50) NOT NULL, 
  PR_IMG_3 VARCHAR2(50) NOT NULL, 
  PR_IMG_4 VARCHAR2(50) NOT NULL, 
  PR_IMG_5 VARCHAR2(50) NOT NULL, 
  PR_TITLE VARCHAR2(50) NOT NULL,
  PR_CATEGORY VARCHAR2(50) NOT NULL,
  PR_REGION VARCHAR2(50) NOT NULL,
  PR_QUALITY VARCHAR2(20) NOT NULL,
  PR_PRICE VARCHAR2(50) NOT NULL,
  PR_DETAIL VARCHAR2(1000) NOT NULL,
  PRIMARY KEY(PR_ID)
);

//CHAT MESSAGE TABLE
CREATE TABLE CHATMESSAGE (
  ID BIGINT AUTO_INCREMENT,
  RECIPIENTNAME VARCHAR(100) NOT NULL,
  SENDERNAME VARCHAR(100) NOT NULL,
  CONTENT VARCHAR(MAX),
  TIME VARCHAR(50) NOT NULL,
  PRIMARY KEY(ID)
);
